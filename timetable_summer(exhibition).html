function setClock() {         var date = new Date();         //var date = new Date("2025-05-05 19:42:00"); // 현재 시간 테스트용         var year = date.getFullYear();         var month = date.getMonth() + 1;         var day = date.getDate();         var weekNames = ["일", "월", "화", "수", "목", "금", "토"];         var weekNames_en = ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"];         var week = weekNames[date.getDay()];         var week_en = weekNames_en[date.getDay()];         var hour = modifyNumber(date.getHours());         var min = modifyNumber(date.getMinutes());         var sec = modifyNumber(date.getSeconds());         var first_arrow_display = "block";         var second_arrow_display = "block";         // 오전, 오후 구분         var am_pm = hour < 12 ? "오전" : "오후";         var displayHour = hour % 12 === 0 ? 12 : hour % 12; // 12시간 형식으로 변환         // 2시 45분 미만 = 관측실(2시 25분)         if (am_pm == "오전" || (am_pm == "오후" && hour < 14) || (hour == 14 && min < 45)) {             current_pg = f3;             current_pg_sub = "14" + f3_time1 + "14" + f3_time2;             current_pg_en = en_f3;             current_pg_en_sub = "2" + en_f3_time1 + "2" + en_f3_time2; // 기존의 2시 표기 유지         }         // 19시 45분 전까지는 관측실 <-------------------------- [수정된 부분 1: 조건 순서 변경]         else if (hour == 19 && min < 45) {             current_pg = f3;             current_pg_sub = "19" + f3_time1 + "19" + f3_time2;             current_pg_en = en_f3;             current_pg_en_sub = "7" + en_f3_time1 + "7" + en_f3_time2; // 7시로 표기         }         // 19시 45분 이상, 19시 58분 미만 = 투영실 <-------------------------- [수정된 부분 2: 조건 순서 변경]         else if (hour == 19 && min >= 45 && min < 58) {             current_pg = f1;             current_pg_sub = "19" + f1_time;             current_pg_en = en_f1;             current_pg_en_sub = en_f1_time1 + "7" + en_f1_time2; // 7시로 표기         }         // 17시 45분 이상, 18시 58분 미만 = 투영실(19시) <-------------------------- [기존 조건 유지]         else if ((hour == 19 && min >= 45) || (hour == 19 && min < 58)) {             // 화요일 20시일 경우 시낭송             if (week == "화") {                 current_pg = f1_poetry_recitation;                 current_pg_sub = "20" + f1_time_concert;                 current_pg_en = en_f1_poetry_recitation;                 current_pg_en_sub = en_f1_time1_concert + "8" + en_f1_time2_concert; // 8시로 표기             }             // 토요일 20시일 경우 음악회             else if (week == "토") {                 current_pg = f1_music_concert;                 current_pg_sub = "20" + f1_time_concert;                 current_pg_en = en_f1_music_concert;                 current_pg_en_sub = en_f1_time1_concert + "8" + en_f1_time2_concert; // 8시로 표기             }             else {                 current_pg = f1;                 current_pg_sub = "19" + f1_time;                 current_pg_en = en_f1;                 current_pg_en_sub = en_f1_time1 + "7" + en_f1_time2; // 7시로 표기             }         }         // 17시 45분 이상, 18시 58분 미만 = 저녁시간(19시)         else if ((hour == 17 && min >= 0) || (hour == 18 && min < 50)) {             // 저녁 18~19시 저녁시간                 current_pg = dinner;                 current_pg_sub = dinner_time;                 current_pg_en = en_dinner;                 current_pg_en_sub = en_dinner_time;         }         // 21시 45분 이상 = 프로그램 X         else if ((hour == 21 && min >= 45) || (hour > 21))         {             current_pg = end;             current_pg_sub = end_sub;             current_pg_en = en_end;             current_pg_en_sub = en_end_sub;         }         else { // 이 else는 매시간 0분 ~ 44분 사이의 기본 관측실 조건이었음             // 매시간 45분 이상, 58분 미만 = 투영실             if (min >= 45 && min < 58) {                 current_pg = f1;                 current_pg_sub = displayHour + f1_time;                 current_pg_en = en_f1;                 current_pg_en_sub = en_f1_time1 + displayHour + en_f1_time2;             }             // 매시간 58분 이상, 00분 미만 = 관측실             else if (min >= 58) {                 current_pg = f3;                 current_pg_sub = (displayHour + 1) + f3_time1 + (displayHour + 1) + f3_time2;                 current_pg_en = en_f3;                 current_pg_en_sub = (displayHour + 1) + en_f3_time1 + (displayHour + 1) + en_f3_time2;             }             else {                 current_pg = f3;                 current_pg_sub = displayHour + f3_time1 + displayHour + f3_time2;                 current_pg_en = en_f3;                 current_pg_en_sub = displayHour + en_f3_time1 + displayHour + en_f3_time2;             }         }         document.getElementById("date").innerHTML = year + "/" + month + "/" + day + " " + week;         document.getElementById("time").innerHTML = am_pm + " " + displayHour + ":" + min + ":" + sec;         document.getElementById("current_pg").innerHTML = current_pg;         document.getElementById("current_pg_sub").innerHTML = current_pg_sub;         document.getElementById("current_pg_en").innerHTML = current_pg_en;         document.getElementById("current_pg_en_sub").innerHTML = current_pg_en_sub;     }     function modifyNumber(time) {         if (parseInt(time) < 10) {             return "0" + time;         }         else             return time;     }     window.onload = function () {         setClock();         setInterval(setClock, 1000); //1초마다 setClock 함수 실행     }
